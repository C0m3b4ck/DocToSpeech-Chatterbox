#
#   Step 2: Install chatterbox without pulling its own deps:
#     pip install chatterbox-tts --no-deps
#
#   Step 3: Install remaining deps:
#     pip install -r requirements-chatterbox.txt
#
#   Then run doctospeech from this venv.
